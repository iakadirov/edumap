## 2. Информация на карточке школы

### Иерархия данных по важности

```
КРИТИЧЕСКИ ВАЖНО (видно сразу):
├── Фото школы
├── Название
├── Рейтинг (звёзды + число)
├── Цена
├── Район
└── Badges (верификация, топ, награды)

ВАЖНО (видно при сканировании):
├── Тип школы
├── Программа (Cambridge, IB...)
├── Классы (1-11)
├── Язык обучения
├── Количество отзывов
└── Ключевые особенности (2-3 тега)

ДОПОЛНИТЕЛЬНО (при hover или клике):
├── Краткое описание
├── Quick actions
└── Кнопки действий
```

## 8. Состояния карточки

### Default

```css
.school-card {
  border: 1px solid #E5E7EB;  /* gray-200 */
  background: #FFFFFF;
  border-radius: 12px;
  box-shadow: none;
}
```

### Hover

```css
.school-card:hover {
  border-color: #BFDBFE;  /* blue-200 */
  box-shadow: 0 4px 6px -1px rgba(0, 0, 0, 0.1);
  transform: translateY(-2px);
  transition: all 0.2s ease;
}
```

### Saved (в избранном)

```
Кнопка ♡ (outline) → ♥ (filled, red-500)
```

### In Comparison (добавлена в сравнение)

```css
.school-card.in-comparison {
  border: 2px solid #3B82F6;  /* blue-500 */
}
/* Чекбокс "Сравнить" → checked */
/* Маленький badge в углу: "В сравнении" */
```

### Premium (платное размещение)

```css
.school-card.premium {
  border: 2px solid transparent;
  background: linear-gradient(white, white) padding-box,
              linear-gradient(135deg, #D97706, #F59E0B) border-box;
}
/* Маленькая звёздочка ⭐ в правом верхнем углу */
```

interface SchoolCard {
  // Основное
  id: string;
  slug: string;

  logo?: string;

  type: 'private' | 'international' | 'state_enhanced';

  // Локация
  district: {
    id: string;
    name: string;
  };
  address?: string;
  
  // Образование
  gradesFrom: number;      // 1
  gradesTo: number;        // 11
  languages: ('russian' | 'uzbek' | 'english')[];
  isBilingual: boolean;
  
  // Цена
  fee: {
    min: number;
    max?: number;
    currency: 'UZS' | 'USD';
    period: 'month' | 'year';
  };
  
  // Рейтинг
  rating: {
    score: number | null;  // 4.8 или null если новая
    reviewCount: number;
    rank?: number;         // #3 в рейтинге
  };
  
  // Статусы и badges
  badges: {
    isVerified: boolean;
    isPremium: boolean;
    isNew: boolean;        // < 6 месяцев на платформе
    hasOpenEnrollment: boolean;
    ranking?: {
      type: 'top1' | 'top3' | 'top10' | 'parents_choice' | 'year_best';
      year?: number;
    };
    awards?: {
      name: string;
      icon: string;
      year: number;
    }[];
  };
  
  // Акции
  promotion?: {
    text: string;
    discount?: number;     // 20 = 20%
    validUntil: Date;
  };
  
  // Особенности (для иконок)
  features: {
    hasPool: boolean;
    hasTransport: boolean;
    hasMeals: boolean;
    hasExtendedDay: boolean;
    hasStem: boolean;
    hasArts: boolean;
    hasMusic: boolean;
    hasSports: boolean;
    hasIT: boolean;
  };
  
  // Краткое описание
  shortDescription?: string;
}

  school: SchoolCard;
  variant?: 'default' | 'compact' | 'featured';
  onCompareToggle?: (id: string) => void;
  onSaveToggle?: (id: string) => void;
  isInComparison?: boolean;
  isSaved?: boolean;
}
